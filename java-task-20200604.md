       HashMap<Integer, String> hezi = new HashMap<>(); //存放扑克牌和它的索引

   //生成扑克牌，从小到大存入集合中，这里采取的花色大小是：梅花<方片<黑桃<红桃

                     String puke = colors[j].concat(numbers[i]);

    list.add(index);
 
  
        index++;

   }

  }


    hezi.put(index, "小王");

                   dpset.add(list.get(i));

   w1set.add(list.get(i));

    w3set.add(list.get(i));

  }
